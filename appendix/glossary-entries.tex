% Glossary entries
\newglossaryentry{apig} {
    name=\glslink{api}{API},
    text=API,
    sort=api,
    description={in informatica con il termine \emph{Application Programming Interface API} (ing. interfaccia di programmazione di un'applicazione) si indica ogni insieme di procedure disponibili al programmatore, di solito raggruppate a formare un set di strumenti specifici per l'espletamento di un determinato compito all'interno di un certo programma. La finalità è ottenere un'astrazione, di solito tra l'hardware e il programmatore o tra software a basso e quello ad alto livello semplificando così il lavoro di programmazione}
}

\newglossaryentry{BIg} {
    name=\glslink{BI}{BI},
    text=BI,
    sort=BI,
    description={\emph{BI, Business Intelligence} (ing. intelligenza aziendale) è un insieme di processi, tecnologie e strumenti utilizzati per raccogliere, analizzare e presentare dati aziendali al fine di supportare il processo decisionale. La BI consente alle organizzazioni di trasformare i dati grezzi in informazioni significative e utili, facilitando l'identificazione di tendenze, modelli e opportunità di business. Gli strumenti di BI includono software di analisi dei dati, dashboard interattivi, reportistica e visualizzazione dei dati, che aiutano le aziende a prendere decisioni informate basate su dati concreti}
}

\newglossaryentry{DBMSg} {
    name=\glslink{DBMS}{DBMS},
    text=DBMS,
    sort=dbms,
    description={\emph{DBMS, Database Management System} (ing. sistema di gestione di database) è un software che consente la creazione, la gestione e l'interrogazione di database. Un DBMS fornisce un'interfaccia per gli utenti e le applicazioni per interagire con i dati memorizzati in un database, consentendo operazioni come l'inserimento, l'aggiornamento, la cancellazione e il recupero di informazioni. I DBMS sono utilizzati in una vasta gamma di applicazioni, dalle piccole applicazioni desktop ai grandi sistemi aziendali e alle piattaforme web}
}

\newglossaryentry{Dockerg} {
    name=\glslink{docker}{Docker},
    text=Docker,
    sort=docker,
    description={\emph{Docker} è una piattaforma software che consente di creare, distribuire e gestire applicazioni in container. Un container è un'unità leggera e portatile che include tutto il necessario per eseguire un'applicazione, come il codice, le librerie, le dipendenze e le configurazioni. Docker semplifica il processo di sviluppo, test e distribuzione delle applicazioni, consentendo agli sviluppatori di creare ambienti isolati e coerenti su qualsiasi sistema operativo o infrastruttura. Docker è ampiamente utilizzato per la gestione di microservizi, l'orchestrazione di container e la creazione di ambienti di sviluppo replicabili}
}

\newglossaryentry{gbg} {
    name=\glslink{gb}{GB},
    text=GB,
    sort=gb,
    description={\emph{Gigabyte} (abbreviato in \emph{GB}) è un'unità di misura della quantità di informazioni digitali. Un gigabyte equivale a 1.073.741.824 byte (2^30 byte) o, in termini decimali, a 1.000.000.000 byte (10^9 byte)}
}

\newglossaryentry{httpg}{
    name=\glslink{http}{HTTP},
    text=HTTP,
    sort=http,
    description={\emph{HTTP, Hypertext Transfer Protocol} (ing. protocollo di trasferimento ipertestuale) è un protocollo di comunicazione utilizzato per la trasmissione di dati su Internet. HTTP è il protocollo alla base del World Wide Web e consente ai client (come i browser web) di comunicare con i server per richiedere e ricevere risorse, come pagine web, immagini e altri contenuti. HTTP è un protocollo stateless, il che significa che ogni richiesta è indipendente dalle altre}
}

\newglossaryentry{IDEg}{
    name=\glslink{IDE}{IDE},
    text=IDE,
    sort=IDE,
    description={\emph{IDE, Integrated Development Environment} (ing. ambiente di sviluppo integrato) è un software che fornisce agli sviluppatori un insieme completo di strumenti per la scrittura, il debug e la gestione del codice sorgente. Un IDE tipicamente include un editor di codice, un compilatore o interprete, un debugger e strumenti per la gestione dei progetti. L'obiettivo di un IDE è semplificare il processo di sviluppo software, migliorare la produttività e facilitare la gestione del ciclo di vita del software}
}

\newglossaryentry{JVMg}{
    name=\glslink{JVM}{JVM},
    text=JVM,
    sort=JVM,
    description={\emph{JVM, Java Virtual Machine} (ing. macchina virtuale Java) è un'implementazione software di una macchina virtuale che esegue programmi scritti in Java. La JVM fornisce un ambiente di esecuzione indipendente dalla piattaforma, consentendo ai programmi Java di essere eseguiti su qualsiasi sistema operativo o hardware che supporti la JVM. La JVM gestisce la memoria, l'esecuzione del codice e altre funzionalità necessarie per eseguire i programmi Java in modo efficiente e sicuro}
}

\newglossaryentry{KPIg}{
    name=\glslink{KPI}{KPI},
    text=KPI,
    sort=KPI,
    description={\emph{KPI, Key Performance Indicator} (ing. indicatore chiave di prestazione) è una metrica utilizzata per valutare le prestazioni di un'organizzazione, di un progetto o di un processo rispetto a obiettivi specifici. I KPI sono strumenti essenziali per monitorare i progressi, identificare aree di miglioramento e prendere decisioni informate. I KPI possono essere quantitativi (ad esempio, il numero di vendite, il tempo di risposta) o qualitativi (ad esempio, la soddisfazione del cliente) e sono spesso utilizzati in ambito aziendale, ma possono essere applicati anche in altri contesti come l'istruzione, la sanità e il settore pubblico}
}

\newglossaryentry{mysqlg} {
    name=\glslink{mysql}{MySQL},
    text=MySQL,
    sort=mysql,
    description={\emph{MySQL} è un sistema di gestione di database relazionali (RDBMS) open source basato su SQL (Structured Query Language). È uno dei database più popolari al mondo, ampiamente utilizzato per applicazioni web e software aziendali. MySQL è noto per la sua velocità, affidabilità e facilità d'uso, ed è spesso utilizzato in combinazione con linguaggi di programmazione come PHP, Python e Java per creare applicazioni dinamiche e interattive}
}

\newglossaryentry{pdfg}{
    name=\glslink{pdf}{PDF},
    text=PDF,
    sort=pdf,
    description={\emph{PDF, Portable Document Format} (ing. formato di documento portatile) è un formato di file sviluppato da Adobe Systems per rappresentare documenti in modo indipendente dall'hardware, dal software e dal sistema operativo utilizzati per crearli o visualizzarli. I file PDF possono contenere testo, immagini, grafica vettoriale e altri elementi multimediali, mantenendo la formattazione originale del documento. Il formato PDF è ampiamente utilizzato per la condivisione di documenti elettronici, in quanto garantisce che il contenuto venga visualizzato correttamente su diverse piattaforme e dispositivi}
}

\newglossaryentry{PoCg}{
    name=\glslink{PoC}{PoC},
    text=PoC,
    sort=PoC,
    description={\emph{PoC, Proof of Concept} (ing. prova di concetto) è una dimostrazione pratica o un prototipo sviluppato per verificare la fattibilità di un'idea, di un concetto o di una tecnologia. Un PoC viene spesso utilizzato per testare l'efficacia di una soluzione proposta prima di investire risorse significative nello sviluppo completo. Il PoC può essere utilizzato in vari contesti, come lo sviluppo software, la sicurezza informatica, l'ingegneria e altri campi, al fine di valutare se un'idea o una tecnologia può essere implementata con successo e soddisfare i requisiti desiderati}
}

\newglossaryentry{restg}{
    name=\glslink{REST}{REST},
    text=REST,
    sort=REST,
    description={\emph{REST, Representational State Transfer API} (ing. trasferimento di stato rappresentazionale) è uno stile architetturale per la progettazione di servizi web. REST si basa su un insieme di principi e vincoli che guidano la progettazione e l'implementazione di API (Application Programming Interface) che consentono la comunicazione tra client e server. Le API RESTful utilizzano i metodi HTTP standard (come GET, POST, PUT, DELETE) per eseguire operazioni sui dati e sono progettate per essere semplici, scalabili e facilmente integrabili con altre applicazioni}
}

\newglossaryentry{sqlg}{
    name=\glslink{sql}{SQL},
    text=SQL,
    sort=sql,
    description={\emph{SQL, Structured Query Language} (ing. linguaggio strutturato di interrogazione) è un linguaggio standardizzato per la gestione e l'interrogazione dei database relazionali. È utilizzato per definire e manipolare i dati in un database, consentendo operazioni come la creazione, l'aggiornamento, la cancellazione e il recupero di informazioni. SQL è ampiamente utilizzato in applicazioni software e sistemi di gestione di database per interagire con i dati in modo efficiente e strutturato}
}

Gran parte della letteratura di settore usa tale linguaggio per descrivere soluzioni analitiche e progettuali in modo sintetico e comprensibile a un vasto pubblico}
